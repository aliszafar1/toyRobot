function getCoordinates(value) {
    value = value.slice(6);
    const x = parseInt(value.slice(0, 1));
    const y = parseInt(value.slice(2, 3));
    const direction = value.slice(4)
    return { x, y, direction }
}
 
function updateCoordinates(state) {
    let {x, y, direction} = state;
    switch (direction) {
        case 'NORTH':
            (y < 4) && (y += 1);
            break;
        case 'SOUTH':
            (y > 0) && (y -= 1);
            break;
        case 'EAST':
            (x < 4) && (x += 1);
            break;
        case 'WEST':
            (x > 0) && (x -= 1);
            break;
    }
    return { x, y, direction }
}
